The problem of finding a suitable path to go from a point A to another point B

```py
    def dijkstra(start) -> dict:

        INF = float('inf')
        paths = {k: INF for k in vertices}; paths[start] = 0
        S = set(); V = {k for k in vertices};

        G, A = transform()
        
        while S != V:
            t = {k: v for k, v in paths.items() if k not in S}
            v = min(t, key = t.get) # type: ignore
            S.add(v)

            for w in G[v]:
                if paths[w] > paths[v] + A[(v, w)]:
                    paths[w] = paths[v] + A[(v, w)]
        
        return paths
```
